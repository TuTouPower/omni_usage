## ESLint 配置

`eslint.config.ts`（flat config）：

```typescript
import tseslint from "typescript-eslint";
import react from "eslint-plugin-react";
import reactHooks from "eslint-plugin-react-hooks";
import jsxA11y from "eslint-plugin-jsx-a11y";
import importX from "eslint-plugin-import-x";
import unicorn from "eslint-plugin-unicorn";
import sonarjs from "eslint-plugin-sonarjs";
import security from "eslint-plugin-security";
import promise from "eslint-plugin-promise";
import node from "eslint-plugin-n";
import perfectionist from "eslint-plugin-perfectionist";

export default tseslint.config(
    ...tseslint.configs.strictTypeChecked,
    ...tseslint.configs.stylisticTypeChecked,
    {
        languageOptions: {
            parserOptions: {
                projectService: true,
                tsconfigRootDir: import.meta.dirname,
            },
        },
    },
    // React
    { plugins: { react }, rules: { "react/jsx-uses-react": "error" } },
    {
        plugins: { "react-hooks": reactHooks },
        rules: {
            "react-hooks/rules-of-hooks": "error",
            "react-hooks/exhaustive-deps": "error",
        },
    },
    { plugins: { "jsx-a11y": jsxA11y } },
    // import
    { plugins: { "import-x": importX }, rules: { "import-x/no-cycle": "error" } },
    // 其他插件
    { plugins: { unicorn } },
    { plugins: { sonarjs } },
    { plugins: { security } },
    { plugins: { promise } },
    { plugins: { n: node } },
    { plugins: { perfectionist } },
    // 关键规则（error 级）
    {
        rules: {
            "@typescript-eslint/no-explicit-any": "error",
            "@typescript-eslint/no-unsafe-assignment": "error",
            "@typescript-eslint/no-unsafe-member-access": "error",
            "@typescript-eslint/no-unsafe-call": "error",
            "@typescript-eslint/no-floating-promises": "error",
            "@typescript-eslint/await-thenable": "error",
            "@typescript-eslint/switch-exhaustiveness-check": "error",
            "@typescript-eslint/consistent-type-imports": "error",
        },
    },
    // 忽略
    {
        ignores: ["dist/", "out/", ".vite/", "node_modules/"],
    },
);
```

运行：`eslint . --max-warnings=0`（warning = error）

## Husky + lint-staged

### Husky 初始化

```bash
pnpm exec husky init
```

### pre-commit hook

`.husky/pre-commit`：

```bash
pnpm exec lint-staged
```

### lint-staged 配置

`package.json` 中：

```json
{
    "lint-staged": {
        "*.{ts,tsx}": ["eslint --fix", "prettier --write"],
        "*.{json,md,css}": ["prettier --write"]
    }
}
```

### pre-push hook

`.husky/pre-push`：

```bash
pnpm typecheck
pnpm test
gitleaks detect --source .
```

```json
{
    "scripts": {
        "typecheck": "tsc --noEmit",
        "lint": "eslint . --max-warnings=0",
        "format:check": "prettier --check .",
        "format": "prettier --write .",
        "deadcode": "knip",
        "arch": "depcruise src --validate .dependency-cruiser.cjs",
        "security:js": "pnpm audit --audit-level=high && gitleaks detect --source .",
        "check": "pnpm typecheck && pnpm lint && pnpm format:check && pnpm deadcode && pnpm arch"
    }
}
```

## 验收标准

- [ ] `pnpm typecheck` exit 0
- [ ] `pnpm lint` exit 0（`--max-warnings=0`）
- [ ] `pnpm format:check` exit 0
- [ ] `pnpm deadcode` exit 0
- [ ] `pnpm arch` exit 0
- [ ] `pnpm check` 一键全绿
- [ ] `git commit` 触发 lint-staged（pre-commit hook）
- [ ] `git push` 触发 typecheck + test + gitleaks（pre-push hook）
- [ ] 无循环依赖
- [ ] renderer 不 import main 或 node:\*
